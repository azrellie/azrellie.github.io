Thermodynamic equations:

Relative Humidity:

Where:

RH = Output

TD = Dew point in Celcius

T = Air temperature in Celcius

exp = Exponent function

RH = 100 * (exp((17.625 * TD) / (243.04 + TD)) / exp((17.625 * T) / (243.04 + T)))

Code conversions:

C++/C


// dew_point and air_temp in C
double rh = Math::Floor(100 * (Math::Exp((17.625 * dew_point) / (243.04 + dew_point)) / Math::Exp((17.625 * air_temp) / (243.04 + air_temp))));

C#


// dew_point and air_temp in C
double rh = Math.Floor(100 * (Math.Exp((17.625 * dew_point) / (243.04 + dew_point)) / Math.Exp((17.625 * air_temp) / (243.04 + air_temp))));

Lua


-- dew_point and air_temp in C
local rh = math.floor(100 * (math.exp((17.625 * dew_point) / (243.04 + dew_point)) / math.exp((17.625 * air_temp) / (243.04 + air_temp))))

Javascript

Dew Point:

Where:

TD = Dew point in Celcius

T = Air temperature in Celcius

RH = Relative Humidity

log = Logarithmic function

TD = 243.04 * (log(RH / 100) + ((17.625 * T) / (243.04 + T))) / (17.625 - log(RH / 100) - ((17.625 * T) / (243.04 + T)))

Code conversions:

C++/C

C#


// air_temp and dew_point in C

Lua

Air Temperature:

Where:

TD = Dew point in Celcius

T = Air temperature in Celcius

RH = Relative Humidity

log = Logarithmic function

T = 243.04 * (((17.625 * TD) / (243.04 + TD)) - log(RH / 100)) / (17.625 + log(RH / 100) - ((17.625 * TD) / (243.04 + TD)))

Code conversions:

Heat Index:

Where:

T = Air temperature in Fahrenheit

RH = Relative Humidity

HI = Heat index in Fahrenheit

HI = -42.379 + 2.04901523 * T + 10.14333127 * RH - 0.22475541 * T * RH - 0.00683783 * T * T - 0.05481717 * RH * RH + 0.00122874 * T * T * RH + 0.00085282 * T * RH * RH - 0.00000199 * T * T * RH * RH

If RH <13% and T > 80 and T < 112 then subtract from HI with this amount

ADJUSTMENT = (13 - RH) / 4 * sqrt((17 - ABS(T - 95)) / 17)

If RH >85% and T > 80 and T < 87 then add from HI with this amount

ADJUSTMENT = ((RH - 85) / 10) * ((87 - T) / 5)

Code conversions:

C++/C


// heat index in fahrenheit
// air_temp in F
double six = 6.83783;
double five = 5.481717;
double one = 1.22874;
double eight = 8.5282;
double temp_p2 = Math::Pow(air_temp, 2);
double rh_p2 = Math::Pow(rh, 2);
double hi = -42.379 + (2.04901523 * air_temp) + (10.14333127 * rh) - (0.22475541 * air_temp * rh) - (six * Math::Pow(10, -3) * temp_p2) - (five * Math::Pow(10, -2) * rh_p2) + (one * Math::Pow(10, -3) * temp_p2 * rh) + (eight * Math::Pow(10, -4) * air_temp * rh_p2) - (1.99 * Math::Pow(10, -6) * temp_p2 * rh_p2);
if (rh <= 13 && air_temp >= 80 && air_temp <= 112)
{
    double sqrt = Math::Sqrt(17 - Math::Abs(air_temp - 95) / 17);
    double num = ((13 - rh) / 4);
    double final_equation = num * sqrt;
    hi -= final_equation;
}
else if (rh >= 85 && air_temp >= 80 && air_temp <= 87)
{
    hi += ((rh - 85) / 10) * ((87 - air_temp) / 5);
}
else if (air_temp < 80)
{
    hi = 0.5 * (air_temp + 61 + ((air_temp - 68) * 1.2) + (rh * 0.094));
}

Specific Humidity:

Where:

TD = Dew point in Celcius

P = Air pressure in hPA or mBar

a = Variablized equation for readability

SH = Specific Humidity in g/kg

a = 6.112 * exp((17.67 * TD) / (TD + 243.5))

SH = (0.622 * a) / (P - (0.378 * a)) * 1000

Code conversions:

C++/C


// dew_point in C
// air_pressure in hPa/mBar
double sh_exp = 6.112 * Math::Exp((17.67 * dew_point) / (dew_point + 243.5));
double sh = (0.622 * sh_exp) / (air_pressure - (0.378 * sh_exp)) * 1000;

C#

Absolute Humidity:

Where:

TD = Air temperature in Celcius

RH = Relative humidity

AH = Absolute Humidity in g/m3

pow = Exponent power function

AH = 6.112 * pow(2.71828, ((17.67 * TD) / (TD + 243.5))) * RH * 2.1674 / (273.15 + TD)

Lapse Rate:

Where:

T = Air temperature in Celcius

EL = Air temperature in Celcius at 12000 feet

a, b, c, d = Variablized equation for readability

exp = Exponent function

LP = Lapse rate in C/km

EL = AirTempAtElevation(T, 12000)

a = 2510 - 2.38 * EL

b = (6.11 / 1012.34) * exp(0.622 * a / 0.287 / 273 / (1 * T + 273) * T)

c = 0.622 * a * b / (0.287 * (1 * T + 273) * (1 * T + 273))

d = 10 * (1 + 0.622 * a * b / (1 * 0.287 * (1 * T + 273))) / (1 + (0.622 * a * c) / (1.005 * 1))

LP = d

Code conversions:

C++/C


// air_temp in C
double FtoC(double f)
{
    return (f - 32) * 5 / 9;
}
double AirTemperatureAtElevation(double air_temp_f, double elevation_feet) // returns in f
{
    return air_temp_f - (3.5 * elevation_feet) / 1000;
}
double elevation = 12000;
double a = 2510 - 2.38 * FtoC(AirTemperatureAtElevation(air_temp, elevation));
double b = (6.11 / 1012.34) * Math::Exp(0.622 * a / 0.287 / 273 / (1 * air_temp + 273) * air_temp);
double c = 0.622 * a * b / (0.287 * (1 * air_temp + 273) * (1 * air_temp + 273));
double d = 10 * (1 + 0.622 * a * b / (1 * 0.287 * (1 * air_temp + 273))) / (1 + (0.622 * a * c) / (1.005 * 1));
double lp = Math::Floor(d * 10) / 10;

CAPE:

Where:

P = Air pressure in hPa or mBar

TD = Dew point in Celcius

VT = Virtual temperature in Kelvin

PT = Air temperature in Kelvin at 12000 feet

VTC = Virtual temperature in Celcius

pow = Exponent power function

VT = (TD + 273.15) / (1 - 0.379 * (6.11 * pow(10, ((7.5 * TD) / (237.7 + TD))) / P))

PT = AirTempAtElevation(VT, 12000)

VTC = KtoC(VT)

CAPE = 9.81 * ((VT - VTC) / VTC)

Code conversions:

C++/C


// parcel_temp will be in F for this case
// air_pressure in hPa/mBar
// air_temp and dew_point in C
// vt_pressure in K
double KtoF(double k)
{
    return (k - 273.15) * 9 / 5 + 32;
}
double FtoK(double f)
{
    return (f - 32) * 5 / 9 + 273.15;
}
double KtoC(double k)
{
    return k - 273.15;
}
double AirTemperatureAtElevation(double air_temp_f, double elevation_feet) // returns in f
{
    return air_temp_f - (3.5 * elevation_feet) / 1000;
}
double vt_pressure = air_pressure;
double virtual_temperature = (air_temp + 273.15) / (1 - 0.379 * (6.11 * Math::Pow(10, ((7.5 * dew_point) / (237.7 + dew_point))) / vt_pressure));
double parcel_temp = AirTemperatureAtElevation(KtoF(virtual_temperature), 20000);
double parcel_temp_k = FtoK(parcel_temp);
double virtual_temperature_c = KtoC(virtual_temperature);
double ca = 9.81 * ((parcel_temp_k - virtual_temperature_c) / virtual_temperature_c);
